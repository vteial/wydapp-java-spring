001    package wyd.dto;
002    
003    import java.util.List;
004    
005    import com.google.common.base.Objects;
006    import com.google.common.base.Objects.ToStringHelper;
007    import com.google.common.base.Strings;
008    
009    public class Role extends AbstractDTO {
010    
011            @Override
012            public void setDefaultValues() {
013                    this.setId("");
014                    this.setName("");
015            }
016    
017            @Override
018            public String toString() {
019                    ToStringHelper toStringHelper = Objects.toStringHelper(this);
020                    toStringHelper.add("id", this.id);
021                    toStringHelper.add("name", this.name);
022                    toStringHelper.add("status", this.status);
023                    toStringHelper.add("operations", this.operations);
024                    toStringHelper.add("createdTime", this.createdTime);
025                    toStringHelper.add("updatedTime", this.updatedTime);            
026                    return toStringHelper.toString();
027            }
028    
029            private String id;
030    
031            public String getId() {
032                    return id;
033            }
034    
035            public void setId(String id) {
036                    this.id = id;
037            }
038    
039            private String name;
040    
041            public String getName() {
042                    return name;
043            }
044    
045            public void setName(String name) {
046                    this.name = name;
047            }
048    
049            private String status;
050    
051            public String getStatus() {
052                    return this.status;
053            }
054    
055            public void setStatus(String status) {
056                    status = Strings.nullToEmpty(status);
057                    status = status.toLowerCase();
058                    if (status.equals(DTO.ENABLED)) {
059                            this.status = DTO.ENABLED;
060                    } else {
061                            this.status = DTO.DISABLED;
062                    }
063            }
064    
065            private String operations;
066    
067            public String getOperations() {
068                    return operations;
069            }
070    
071            public void setOperations(String operations) {
072                    this.operations = operations;
073            }
074    
075            private List<User> users;
076    
077            public List<User> getUsers() {
078                    return users;
079            }
080    
081            public void setUsers(List<User> users) {
082                    this.users = users;
083            }
084    }
