001    package wyd.dto;
002    
003    import com.google.common.base.Objects;
004    import com.google.common.base.Objects.ToStringHelper;
005    import com.google.common.base.Strings;
006    
007    public class User extends AbstractDTO {
008    
009            @Override
010            public void setDefaultValues() {
011                    this.setId("");
012                    this.setPassword("");
013                    this.setName("");
014                    this.setEmailId("");
015                    this.setStatus(DTO.ENABLED);
016            }
017    
018            @Override
019            public String toString() {
020                    ToStringHelper toStringHelper = Objects.toStringHelper(this);
021                    toStringHelper.add("id", this.id);
022                    toStringHelper.add("password", this.password);
023                    toStringHelper.add("name", this.name);
024                    toStringHelper.add("emailId", this.emailId);
025                    toStringHelper.add("status", this.status);
026                    if (this.role != null) {
027                            toStringHelper.add("roleId", this.role.getId());
028                    } else {
029                            toStringHelper.add("roleId", "NULL");
030                    }
031                    toStringHelper.add("createdTime", this.createdTime);
032                    toStringHelper.add("updatedTime", this.updatedTime);
033                    return toStringHelper.toString();
034            }
035    
036            private String id;
037    
038            public String getId() {
039                    return id;
040            }
041    
042            public void setId(String id) {
043                    this.id = id;
044            }
045    
046            private String password;
047    
048            public String getPassword() {
049                    return password;
050            }
051    
052            public void setPassword(String password) {
053                    this.password = password;
054            }
055    
056            private String name;
057    
058            public String getName() {
059                    return name;
060            }
061    
062            public void setName(String name) {
063                    this.name = name;
064            }
065    
066            private String emailId;
067    
068            public String getEmailId() {
069                    return emailId;
070            }
071    
072            public void setEmailId(String emailId) {
073                    this.emailId = emailId;
074            }
075    
076            private String parentId;
077    
078            public String getParentId() {
079                    return parentId;
080            }
081    
082            public void setParentId(String parentId) {
083                    this.parentId = parentId;
084            }
085    
086            private String status;
087    
088            public String getStatus() {
089                    return this.status;
090            }
091    
092            public void setStatus(String status) {
093                    status = Strings.nullToEmpty(status);
094                    status = status.toLowerCase();
095                    if (status.equals(DTO.ENABLED)) {
096                            this.status = DTO.ENABLED;
097                    } else {
098                            this.status = DTO.DISABLED;
099                    }
100            }
101    
102            private Role role;
103    
104            public Role getRole() {
105                    return role;
106            }
107    
108            public void setRole(Role role) {
109                    this.role = role;
110            }
111    }
