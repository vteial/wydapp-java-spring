001    package wyd.web;
002    
003    import javax.servlet.http.HttpSession;
004    
005    import org.springframework.beans.factory.annotation.Autowired;
006    import org.springframework.beans.factory.annotation.Qualifier;
007    import org.springframework.stereotype.Controller;
008    import org.springframework.web.bind.annotation.RequestMapping;
009    import org.springframework.web.bind.annotation.RequestParam;
010    import org.springframework.web.bind.annotation.ResponseBody;
011    import org.springframework.web.servlet.ModelAndView;
012    
013    import wyd.dto.User;
014    import wyd.dto.WebMessage;
015    import wyd.service.AcmService;
016    import wyd.service.InvalidCredentialException;
017    
018    @Controller
019    public class WydController extends AbstractController {
020    
021            @Autowired
022            @Qualifier("memoryAcmService")
023            private AcmService acmService;
024    
025            @RequestMapping("/")
026            public ModelAndView home() {
027                    return this.index();
028            }
029    
030            @RequestMapping("/wydIndex.java")
031            public ModelAndView index() {
032                    ModelAndView mav = this.createDefaultModelAndView("wydIndex", "Home");
033    
034                    return mav;
035            }
036    
037            @RequestMapping("/wydLogin.java")
038            public ModelAndView login() {
039                    ModelAndView mav = this
040                                    .createDefaultModelAndView("wydLogin", "Sign In");
041    
042                    return mav;
043            }
044    
045            @RequestMapping("/wydAuthenticate.java")
046            public String authenticate(
047                            @RequestParam("userId") String userId,
048                            @RequestParam("password") String password,
049                            @RequestParam(value = "callbackUrl", required = false) String callbackUrl,
050                            HttpSession session) {
051                    String rstring = null;
052                    try {
053                            User user = acmService.authenticate(userId, password);
054                            session.setAttribute("user", user);
055    
056                            if (callbackUrl != null) {
057                                    rstring = "redirect:" + callbackUrl;
058                            } else {
059                                    rstring = "redirect:wydIndex.java";
060                            }
061    
062                    } catch (InvalidCredentialException ice) {
063                            WebMessage webMessage = WebMessage.createErrorMessage(ice
064                                            .getMessage());
065                            rstring = "redirect:wydLogin.java?" + webMessage.toQueryString();
066                    }
067    
068                    return rstring;
069            }
070    
071            @RequestMapping("/wydLogout.java")
072            public ModelAndView logout(HttpSession session) {
073    
074                    session.removeAttribute("user");
075                    ModelAndView mav = this.index();
076    
077                    return mav;
078            }
079    
080            @RequestMapping("/wydAbout.java")
081            public ModelAndView about() {
082                    ModelAndView mav = this.createDefaultModelAndView("wydAbout", "About");
083    
084                    return mav;
085            }
086    
087            @RequestMapping("/t/ping.java")
088            public @ResponseBody
089            String ping() {
090                    return "ping pong!";
091            }
092    
093    }
