001    package wyd.web;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import javax.servlet.http.HttpServletResponse;
005    import javax.servlet.http.HttpSession;
006    
007    import org.slf4j.Logger;
008    import org.slf4j.LoggerFactory;
009    import org.springframework.web.servlet.handler.HandlerInterceptorAdapter;
010    
011    import wyd.dto.WebMessage;
012    
013    public class GenericSpringInterceptor extends HandlerInterceptorAdapter {
014    
015            final Logger _log = LoggerFactory.getLogger(GenericSpringInterceptor.class);
016    
017            @Override
018            public boolean preHandle(HttpServletRequest req, HttpServletResponse res,
019                            Object handler) throws Exception {
020                    WebMessage webMessage = WebMessage.createMessage(req);
021                    if (webMessage != null) {
022                            req.setAttribute("webMessage", webMessage);
023                            if (_log.isDebugEnabled()) {
024                                    _log.debug("WebMessage = {}", webMessage.toQueryString());
025                            }
026                    }
027    
028                    String callbackUri = req.getParameter("callbackUrl");
029                    if (callbackUri != null) {
030                            req.setAttribute("callbackUrl", callbackUri);
031                            if (_log.isDebugEnabled()) {
032                                    _log.debug("Callback URI = {}", callbackUri);
033                            }
034                    }
035    
036                    HttpSession session = req.getSession(true);
037    
038                    return super.preHandle(req, res, handler);
039            }
040    }
