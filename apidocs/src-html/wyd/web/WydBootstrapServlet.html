001    package wyd.web;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.util.Iterator;
006    import java.util.Properties;
007    
008    import javax.servlet.ServletConfig;
009    import javax.servlet.ServletContext;
010    import javax.servlet.ServletException;
011    import javax.servlet.http.HttpServlet;
012    
013    import org.slf4j.Logger;
014    import org.slf4j.LoggerFactory;
015    
016    public class WydBootstrapServlet extends HttpServlet {
017    
018            private static final long serialVersionUID = 1L;
019    
020            final Logger _log = LoggerFactory.getLogger(WydBootstrapServlet.class);
021    
022            @Override
023            public void init(ServletConfig config) throws ServletException {
024                    super.init(config);
025                    _log.info("################################################");
026                    _log.info("Bootstrap init work starts...");
027    
028                    ServletContext scontext = this.getServletContext();
029    
030                    String s = scontext.getRealPath("/bootloader");
031                    _log.info("realPath : {}", s);
032                    s = s.substring(0, s.indexOf("bootloader"));
033                    scontext.setAttribute("webHome", s);
034                    _log.info("webHome  : {}", scontext.getAttribute("webHome"));
035    
036                    s = (String) scontext.getAttribute("webHome");
037                    s += "WEB-INF/classes/application.properties";
038                    this.loadPropertiesInToServletContext(scontext, new File(s));
039    
040                    _log.info("Bootstrap init work ends...");
041                    _log.info("################################################");
042            }
043    
044            private void loadPropertiesInToServletContext(ServletContext context,
045                            File propsFile) {
046                    try {
047                            Properties props = new Properties();
048                            props.load(new FileInputStream(propsFile));
049                            Iterator i = props.keySet().iterator();
050                            while (i.hasNext()) {
051                                    String key = (String) i.next();
052                                    Object val = props.get(key);
053                                    context.setAttribute(key, val);
054                            }
055                    } catch (Throwable t) {
056                            _log.error("loading application properties failed...", t);
057                    }
058            }
059    }
