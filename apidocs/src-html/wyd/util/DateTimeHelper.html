001    package wyd.util;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Calendar;
005    import java.util.Date;
006    
007    public class DateTimeHelper {
008    
009            public static long toMilliSeconds(final java.util.Date inDate) {
010                    return inDate.getTime();
011            }
012    
013            public static long toMilliSeconds(final java.sql.Date inDate) {
014                    return inDate.getTime();
015            }
016    
017            public static java.util.Date toDateWithTimeOff(final java.util.Date inDate) {
018                    Calendar inCal = Calendar.getInstance();
019                    inCal.setTimeInMillis(inDate.getTime());
020                    inCal.set(Calendar.HOUR_OF_DAY, 0);
021                    inCal.set(Calendar.MINUTE, 0);
022                    inCal.set(Calendar.SECOND, 0);
023                    inCal.set(Calendar.MILLISECOND, 0);
024                    return inCal.getTime();
025            }
026    
027            public static java.sql.Date toSqlDate(final java.util.Date inDate) {
028                    java.sql.Date outDate = new java.sql.Date(inDate.getTime());
029                    return outDate;
030            }
031    
032            public static java.sql.Date toSqlDateWithTimeOff(final java.util.Date inDate) {
033                    Calendar inCal = Calendar.getInstance();
034                    inCal.setTimeInMillis(inDate.getTime());
035                    inCal.set(Calendar.HOUR_OF_DAY, 0);
036                    inCal.set(Calendar.MINUTE, 0);
037                    inCal.set(Calendar.SECOND, 0);
038                    inCal.set(Calendar.MILLISECOND, 0);
039                    java.sql.Date outDate = new java.sql.Date(inCal.getTimeInMillis());
040                    return outDate;
041            }
042    
043            public static java.sql.Date toSqlDateWithTimeOff(final Calendar inCal) {
044                    inCal.set(Calendar.HOUR_OF_DAY, 0);
045                    inCal.set(Calendar.MINUTE, 0);
046                    inCal.set(Calendar.SECOND, 0);
047                    inCal.set(Calendar.MILLISECOND, 0);
048                    java.sql.Date outDate = new java.sql.Date(inCal.getTimeInMillis());
049                    return outDate;
050            }
051    
052            public static java.sql.Timestamp toSqlTimestamp(final java.util.Date inDate) {
053                    return new java.sql.Timestamp(inDate.getTime());
054            }
055    
056            public static java.sql.Timestamp toSqlTimestamp(final java.sql.Date inDate) {
057                    return new java.sql.Timestamp(inDate.getTime());
058            }
059    
060            public static String toDateAsString(SimpleDateFormat sdf, Date date) {
061                    return sdf.format(date);
062            }
063    
064            public static Date getFirstDateOfCurrentMonth() {
065                    Calendar calendar = Calendar.getInstance();
066                    calendar.set(Calendar.DATE,
067                                    calendar.getActualMinimum(Calendar.DAY_OF_MONTH));
068                    return calendar.getTime();
069            }
070    
071            public static Date getLastDateOfCurrentMonth() {
072                    Calendar calendar = Calendar.getInstance();
073                    calendar.set(Calendar.DATE,
074                                    calendar.getActualMaximum(Calendar.DAY_OF_MONTH));
075                    return calendar.getTime();
076            }
077    }
